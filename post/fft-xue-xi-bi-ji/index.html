常数优化

不要写递归形式的，太慢了。

在分治递归时，我们不断将 aia_iai​ 按奇偶排序。设 aia_iai​ 最后拍到的下标是 revirev_irevi​ 。

你随便试几个i。

比如 5(101) 会换到 5(101)。

6(110) 会换到 3(011)

1(001) 会换到 4(100)

会发现把 iii 反转一下就是 revirev_irevi​ 了。

revirev_irevi​ 也可以 O(n log n)O(n\ log\ n)O(n log n) 求了，但是有更快的 O(n)O(n)O(n) 的方法。

revi=(revi>>1>>1)∣((i&1)<<(bit−1))rev_i=(rev_{i>>1}>>1)|((i\&1)<<(bit-1))revi​=(revi>>1​>>1)∣((i&1)<<(bit−1)) 。

翻译成人话就是，iii 的最后一位接上 iii 的前 k−1k-1k−1 位的反转结果。

然后只对 arevia_{rev_i}arevi​​ 操作就可以了。

#include<bits/stdc++.h>
using namespace std;
const double Pi=3.1415926535897932384626;
int rev[262200],bit,tot,n,m;
struct Complex
{
	double x,y;
	Complex operator+(const Complex&n2){return{x+n2.x,y+n2.y};}
	Complex operator-(const Complex&n2){return{x-n2.x,y-n2.y};}
	Complex operator*(const Complex&n2){return{x*n2.x-y*n2.y,x*n2.y+y*n2.x};}
}a[262200],b[262200];
void fft(Complex c[],int inv)
{
	for(int i=0;i<tot;i++)
	if(i<rev[i])swap(c[i],c[rev[i]]);
	for(int mid=1;mid<tot;mid<<=1){
		Complex w1={cos(Pi/mid),inv*sin(Pi/mid)};
		for(int i=0;i<tot;i+=mid*2){
			Complex wk={1,0};
			for(int j=0;j<mid;j++,wk=wk*w1){
				Complex x=c[i+j],y=wk*c[i+j+mid];
				c[i+j]=x+y;
				c[i+j+mid]=x-y;
			}
		}
	}
}
int main()
{
	cin>>n>>m;
	for(int i=0;i<=n;i++)scanf("%lf",&a[i].x);
	for(int i=0;i<=m;i++)scanf("%lf",&b[i].x);
	while((1<<bit)<n+m+1)bit++;
	tot=1<<bit;
	for(int i=0;i<tot;i++)
		rev[i]=(rev[i>>1]>>1)|((i&1)<<(bit-1));
	fft(a,1);fft(b,1);
	for(int i=0;i<tot;i++)a[i]=a[i]*b[i];
	fft(a,-1);
	for(int i=0;i<=n+m;i++)
		printf("%d%c",int(a[i].x/tot+0.5)," \n"[i==n+m]);
}
